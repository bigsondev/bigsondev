import { Post, Text, CodeEditor, Link } from '~components';
import { getIframeContent } from '~utils';

export const tasksListData = [
  <Text size="small" color="#FAFAFA">
    a element exists
  </Text>,
  <Text size="small" color="#FAFAFA">
    Open Me text is inside of the a element
  </Text>,
  <Text size="small" color="#FAFAFA">
    Link opens in a new tab
  </Text>,
  <Text size="small" color="#FAFAFA">
    Link navigates to an external page (start from https:// to open external
    page, e.g. https://facebook.com)
  </Text>,
];
export const helpListData = [
  <Link size="small" href="https://www.w3schools.com/html/html_links.asp">
    Links
  </Link>,
];

export const tabs = ['html'];
export const tests = [
  () => getIframeContent().find('a').length > 0,
  () => {
    const link = getIframeContent().find('a');
    if (!link) {
      return false;
    }
    return link.text().toLowerCase().trim() === 'open me';
  },
  () => getIframeContent().find('a').attr('target') == '_blank',
  () => getIframeContent().find('a').attr('href'),
];

<Post>
  <CodeEditor
    tasks={tasksListData}
    help={helpListData}
    tabs={tabs}
    tests={tests}
    result={`"Open Me" link is displayed on the screen, opens in a new tab and navigates to an external page.`}
  />
</Post>
