export const tabs = ['js'];
export const tests = [
  () => getJs().unique,
  () => getJs().unique?.length === 1,
  () => {
    const unique = getJs().unique?.([1, 2, 3, 4, 5, 1, 2, 3]);
    const expectedResult = [1, 2, 3, 4, 5];
    return (
      unique?.length === 5 &&
      unique.every((value, index) => value === expectedResult[index])
    );
  },
  () => {
    const unique = getJs().unique?.(['bronze', 'silver', 'gold', 'bronze']);
    const expectedResult = ['bronze', 'silver', 'gold'];
    return (
      unique?.length === 3 &&
      unique.every((value, index) => value === expectedResult[index])
    );
  },
];
