import {
  SyntaxHighlighter,
  Link,
  CodeEditorRaw,
  KnowledgePill,
} from '~components';

export const outlineCode = `* {
  outline: 1px solid red;  
}`;

export const tabs = ['html', 'css'];

export const initialHtml = `<div>Hello there, I'm a block element.</div>
<span>Hello div, I'm inline.</span>`;

export const initialCss = `* {
  outline: 1px solid red;
}\n
body {
  height: 100vh;
}\n
div {
  margin-bottom: 1rem;
}`;

<KnowledgePill
  code={
    <SyntaxHighlighter language="css" showCopy>
      {outlineCode}
    </SyntaxHighlighter>
  }
  explanation={
    <>
      <Link
        href="https://www.w3schools.com/cssref/sel_all.asp"
        size="preNormal"
      >
        Select all selector
      </Link>{' '}
      targets all HTML elements and adds{' '}
      <Link
        href="https://www.w3schools.com/css/css_outline.asp"
        size="preNormal"
      >
        outline
      </Link>{' '}
      to them.
    </>
  }
  interactiveExample={
    <CodeEditorRaw
      tabs={tabs}
      initialHtml={initialHtml}
      initialCss={initialCss}
    />
  }
  {...props}
  {...props.frontmatter}
/>
