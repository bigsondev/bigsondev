export const useCounterCode = `import { useState } from "react";\n
export const useCounter = (initialCounter = 0) => {
  const [counter, setCounter] = useState(initialCounter);\n
  const handleIncrement = () => {
    setCounter((prevCounter) => prevCounter + 1);
  };\n
  const handleDecrement = () => {
    setCounter((prevCounter) => prevCounter - 1);
  };\n
  return {
    counter,
    isDecrementDisabled: counter === 0,
    handleIncrement,
    handleDecrement
  };
};
`;

export const appCode = `import React from "react";\n
import { useCounter } from "./useCounter";\n
export default function App() {
  const {
    counter,
    isDecrementDisabled,
    handleIncrement,
    handleDecrement
  } = useCounter(10);\n
  return (
    <div>
      <h1>Simple React Counter</h1>
      <button onClick={handleIncrement}>+</button>
      <h2>{counter}</h2>
      <button onClick={handleDecrement} disabled={isDecrementDisabled}>
        -
      </button>
    </div>
  );
}
`;
