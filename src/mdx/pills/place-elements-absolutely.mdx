export const cssCode = `.container {
  position: relative;
  margin: 0 auto;
}\n
.title {
  position: absolute;
  top: 5%;
  left: 50%;
  transform: translate(-50%);
  margin: 0;
  color: #fff;
}\n
.catImage {
  width: 100%;
  height: auto;
}`;

export const htmlCode = `<div class="container">
  <img src="https://cataas.com/cat/cute" class="catImage" />
  <h2 class="title">I'm cute</h2>
</div>`;
