export const codeSnippet = `// .includes method is also available for String type
const sentence = "All animals are cute but I love cats and dogs the most. ❤️";\n
// We can search for a particular character, word or even sentence
const isCatLover = sentence.includes('cat');
const isDogLover = sentence.includes('and dogs');
const isSpiderLover = sentence.includes('spiders');\n
// We can also drill down the search to start from particular position
const searchFromStart = sentence.includes('All', 0);
const searchFromSecondWord = sentence.includes('All', 4);\n
console.log(isCatLover); // true
console.log(isDogLover); // true
console.log(isSpiderLover); // false\n
console.log(searchFromStart); // true
console.log(searchFromSecondWord); // false`;
