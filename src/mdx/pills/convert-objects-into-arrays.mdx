export const codeSnippet = `const obj = {
  a: 1,
  b: 2,
  c: 3
};\n
// All of the below are now an array,
// you can use all the array methods and properties, 
// e.g. .reduce or .length
const keys = Object.keys(obj); // ['a', 'b', 'c']
const values = Object.values(obj); // [1, 2, 3]
const entries = Object.entries(obj); // [["a", 1], ["b", 2], ["c", 3]]`;
export const initialHtml = '<h1>Open console in your browser to see logs</h1>';

const keys = Object.keys(obj);
const values = Object.values(obj);
const entries = Object.entries(obj);\n
console.log(keys); // ['a', 'b', 'c']
console.log(values); // [1, 2, 3]
console.log(entries); // [["a", 1], ["b", 2], ["c", 3]]`;
